编一C程序，它能读入一串（n个）整数（n<1000，以-9999为结束标记），并判断第n个整数

（即-9999的前一个）在前（n-1）个整数中出现的次数，再输出该次数。（输入时，两个

相邻的整数用空格隔开。）

（注：程序的可执行文件名必须是e2.exe，存于你的账号或其debug目录下，否则无成绩）

#include<stdio.h>
#define MAX 100

void main()
{
	int a[MAX];
	int i=0,c=0,d=0,n,k,j,t;
	printf("判断第n个整数在前（n-1）个整数中出现的次数和在整串出现的总次数：\n");
	printf("请输入第n个整数，再输入一串整数，以-9999为结束标记：\n");
	do{
		scanf("%d",&a[i]);
		if(a[i]==-9999)
			break;
		i++;
	}while(9);
	k=i;
	printf("\n判断第%d个数：\n",a[0]);
	n=a[0];
	for(j=1;j<n;j++){
		if(a[j]==a[n])
			c++;
	}
	t=a[0]-1;
	printf("第%d个整数在前%d个数中出现的次数为：%d\n",a[0],t,c);
	for(i=1;i<k;i++){
		if(a[i]==a[n])
			d++;
	}
	printf("第%d个整数在整串中出现的总次数为：%d\n",a[0],d);
}
